def sample(seed=1,n=1000,tactics=None,
           score=COCOMO2,what='effort'):  
  rseed(seed)
  tactics = tactics or {}
  projects=[anything,flight,ground,osp,osp2]
  COL()
  the.COL.buffer=n
  logAll=N()
  logs= {} 
  for project in projects:
    log = logs[project.__name__] = N()
    for _ in xrange(n):
      settings = complete(project,tactics or {})
      x = score(settings)
      log.tell(x)
      logAll.tell(x)
  print("\n\n---|",what,"|---------------\n")
  for project in projects:
    name = project.__name__
    log = logs[name]
    print ('%9s,' % name,
           xtile(log.kept(),
                 lo=logAll.lo,
                 hi=logAll.hi,
                 chops=[0,0.25,0.5,0.75,0.999],
                 marks=["-"," "," ","-"," "],
                 width=30,
                 show= "%5d"))

@go
def _coc(): 
  sample()
